***************************************
* Contents of the FLOAT directory:
***************************************
 readme.txt: 	To indicate the contents of the Float directory


****************	Subroutines of Floating Point Library	**********************
 Fpinit.asm: 	Floating point library initialization subroutine
		(Fpinit)
 Fpadd.asm: 	Floating point addition subroutine
		(Fadd_xa,Fadd_xy) 
 Fpsub.asm: 	Floating point subtracter subroutine
		(Fsub_xa,Fsub_xy) 
 Fpcmp.asm: 	Floating point comparison subroutine
		(Fcmp_xa,Fcmp_xy) 
 Fpmpy.asm: 	Floating point multiplication subroutine
		(Fmpy_xa,Fmpy_xy) 
 Fpmac.asm: 	Floating point multiply/accumulation subroutine
		(Fmac_xya,Fmac_mxya) 
 Fpdiv.asm: 	Floating point division subroutine
		(Fdiv_xa,Fdiv_xy) 
 Fpscale.asm: 	Floating point scaling subroutine
		(Fscal_xa,Fscal_xy) 
 Fpsqrt.asm: 	Floating point SQRT subroutine
		(Fsqrt_a,Fsqrt_x) 
 Fpneg.asm: 	Floating point negative subroutine
		(Fneg_a,Fneg_x) 
 Fpabs.asm: 	Floating point absolute subroutine
		(Fabs_a,Fabs_x) 
 Fpfix.asm: 	Floating point to fix point coversion subroutine
		(Fix_a,Fix_x) 
 Fpfloat.asm: 	Fix point to floating point coversion subroutine
		(Float_a,Float_x)  
 Fpfloor.asm: 	Floating point FLOOR subroutine
		(Floor_a,Floor_x)  
 Fpceil.asm: 	Floating point CEIL subroutine
		(Ceil_a,Ceil_x)  
 Fpfrac.asm: 	Floating point Fraction subroutine
		(Frac_a,Frac_x)  
**********************************************************************************

 Durbin.asm: 	DURBIN's recursive solution of linear predictive coefficients (LPC)
		using the floating point library
 Fplist.asm:	List of the floating point library


 Fpdef-hlp.txt: 	Help file for defining the storage format and arithmetic representation 
		used by the DSP56300 floating point software subroutine library   
 Fpcalls-hlp.txt: 	Help file for defining the subroutine calling conventions and condition 
		codes used by the DSP56300 floating point software subroutine library 
 Fprevs-hlp.txt:	Help file for defining the filenames, latest version numbers and latest 
		revision date 
 Durbin-hlp.txt: 	Help file for Durbin.asm
